\section{Theoretical Foundation}
%==============================================================================

\subsection{Fundamental Axioms of Scattering Theory}

\begin{axiom}[Fermi's Golden Rule]
The transition rate from initial state $|i\rangle$ to final state $|f\rangle$ due to a perturbation $V$ is:
\begin{equation}
W_{i\to f} = \frac{2\pi}{\hbar}|\langle f|V|i\rangle|^2 \delta(E_f - E_i)
\end{equation}
\end{axiom}

\begin{axiom}[Born Approximation]
For weak scattering potentials, the scattering amplitude is proportional to the Fourier transform of the potential:
\begin{equation}
f(\theta) \propto \int V(\mathbf{r}) e^{i\mathbf{q}\cdot\mathbf{r}} d^3r
\end{equation}
where $\mathbf{q} = \mathbf{k}' - \mathbf{k}$ is the momentum transfer.
\end{axiom}

\begin{axiom}[Linear Response Theory]
The response of an electron gas to an external perturbation is characterized by the dielectric function $\varepsilon(\mathbf{q}, \omega)$.
\end{axiom}

\subsection{Coulomb Scattering by Ionized Impurities}

\begin{definition}[Bare Coulomb Potential]
An ionized impurity of charge $Ze$ creates a potential:
\begin{equation}
V_{\text{bare}}(r) = \frac{Ze^2}{4\pi\varepsilon_0\varepsilon_r r}
\end{equation}
\end{definition}

\begin{theorem}[Fourier Transform of Coulomb Potential]
\begin{equation}
V_{\text{bare}}(q) = \frac{Ze^2}{\varepsilon_0\varepsilon_r q^2}
\end{equation}
\end{theorem}

\subsection{Screening in an Electron Gas}

\begin{definition}[Screened Potential]
In the presence of mobile carriers, the effective potential is:
\begin{equation}
V_{\text{scr}}(q) = \frac{V_{\text{bare}}(q)}{\varepsilon(q)}
\end{equation}
where $\varepsilon(q)$ is the static dielectric function.
\end{definition}

\subsection{Thomas-Fermi Screening}

\begin{theorem}[Thomas-Fermi Dielectric Function]
In the Thomas-Fermi approximation:
\begin{equation}
\varepsilon_{\text{TF}}(q) = \varepsilon_r\left(1 + \frac{q_{\text{TF}}^2}{q^2}\right)
\end{equation}
where the Thomas-Fermi screening wavevector is:
\begin{equation}
q_{\text{TF}} = \sqrt{\frac{k_F m^* e^2}{\pi^2 \hbar^2 \varepsilon_0 \varepsilon_r}}
\end{equation}
\end{theorem}

\begin{definition}[Fermi Wavevector]
For a 3D electron gas with density $n$:
\begin{equation}
k_F = (3\pi^2 n)^{1/3}
\end{equation}
\end{definition}

\subsection{Random Phase Approximation (RPA)}

\begin{theorem}[Lindhard Dielectric Function]
The RPA (Lindhard) dielectric function is:
\begin{equation}
\varepsilon_{\text{RPA}}(q) = \varepsilon_r + \frac{r_s}{\xi}\left[\frac{1}{x} + \frac{1 - x^2/4}{2x^2}\ln\left|\frac{x+2}{x-2}\right|\right]
\end{equation}
where $x = q/k_F$ is the dimensionless wavevector.
\end{theorem}

\begin{definition}[Wigner-Seitz Radius]
The dimensionless density parameter is:
\begin{equation}
r_s = \left(\frac{3}{4\pi n}\right)^{1/3} \frac{m^* e^2}{4\pi\varepsilon_0\hbar^2}
\end{equation}
\end{definition}

\subsection{Scattering Rate Formula}

\begin{theorem}[Differential Scattering Rate]
The angular-dependent scattering rate is:
\begin{equation}
\frac{1}{\tau(\theta)} = \frac{2\pi m^*}{\hbar^3 k^3} n_i \left(\frac{e^2}{4\pi\varepsilon_0}\right)^2 \frac{1}{\varepsilon(q)^2} \cdot \frac{d\eta}{\eta^3}
\end{equation}
where:
\begin{itemize}
\item $\eta = \sin(\theta/2)$
\item $q = 2k\sin(\theta/2)$ is the momentum transfer
\item $n_i$ is the impurity density
\end{itemize}
\end{theorem}

\section{Numerical Method Details}
%==============================================================================

\subsection{Discretization Scheme}

The angular integration is discretized uniformly:
\begin{equation}
\theta_j = \frac{\pi j}{1800}, \quad j = 1, 2, \ldots, 1800
\end{equation}

This gives 0.1¬∞ resolution, sufficient to capture the sharp forward-scattering peak.

\subsection{Lindhard Function Evaluation}

The logarithmic term in the Lindhard function requires care:
\begin{equation}
\ln\left|\frac{x+2}{x-2}\right|
\end{equation}

\begin{itemize}
\item For $x < 2$ (small-angle scattering): Regular evaluation
\item At $x = 2$ (backscattering at $\theta = 180¬∞$): Singularity (pole)
\item For $x > 2$: Argument becomes negative inside, hence absolute value
\end{itemize}

\section{Physical Interpretation}
%==============================================================================

\subsection{TF vs RPA Comparison}

\begin{itemize}
\item \textbf{Thomas-Fermi}: Local approximation, accurate for $q \ll k_F$
\item \textbf{RPA}: Includes non-local response, accurate for all $q$
\end{itemize}

At high carrier density ($n = 10^{18}$ cm$^{-3}$), both models should agree reasonably well because:
\begin{equation}
q_{\text{TF}} \sim k_F \quad \text{(strong screening)}
\end{equation}

\subsection{Angular Dependence}

The scattering rate diverges at small angles:
\begin{equation}
\frac{1}{\tau} \propto \frac{1}{\sin^3(\theta/2)}
\end{equation}

This is regularized by screening (the $q_{\text{TF}}^2$ or RPA terms in the denominator).

%==============================================================================
\section{Summary}
%==============================================================================

This code compares two models for ionized impurity scattering:
\begin{enumerate}
\item Thomas-Fermi: Simple analytical model assuming local screening
\item RPA (Lindhard): More accurate model including the full wavevector-dependent response
\end{enumerate}

At $n = 10^{18}$ cm$^{-3}$, the system is degenerate and strongly screened, so both models give similar results with the characteristic forward-scattering peak regularized by screening.
