%Chapt8Fig14a.m%elastic ionized impurity scattering rate using TF screeing%m=electron mass, n=number of charge impuritiesfunction TFelasticScatteringclear;clf;  m0=9.109382e-31;				%bare electron mass (kg)echarge=1.6021764e-19;		%electron charge (C)hbar=1.05457159e-34;			%Planck's constant (J s)epsilon0=8.8541878e-12;		%permittivity of free-space (F m-1) ezero=13.2;						%low-frequency relative dielectric constantm1=0.070;						%effective electron massemax=300.0d-3;					%maximum particle energy (eV)n=1.0d16;						%carrier density, n (cm-3)npoints = 200;					%number of points in energy plot  for k=1:2   n=n*10;						%plot two curves n=10^17 cm-3 and n=10^18 cm-3 kF1=(3*(pi^2)*n)^(1/3);	%Fermi wave vector (cm-1) kF1=kF1*1.e2;					%convert to units of m-1    kF2=kF1^2;    kF3=kF1^3; qTF2=kF1*m0*m1*(echarge^2)/(epsilon0*ezero*(pi^2)*(hbar^2));	% Thomas-Fermi screening number squared eF=((hbar*kF1)^2)/(2.0*m0*m1*echarge);								% Fermi energy in eV const=(pi*n*1.0e6*(echarge^4)*m0*m1); const=const/(((4.*pi*epsilon0)^2)*(ezero^2)*(hbar^3));  emin=eF;						%initial particle energy demax=(emax-emin)/200.0;	%increment in particle energy (eV) dth=pi/180.0;					%increment in scattering angle theta ei=emin; for i=1:npoints				% the main loop   ei=ei+demax;   ak=(sqrt(2.*m0*m1*ei*echarge))/hbar;      ak3=ak^3;   aint1=0.0;   theta=0.0;  for j=1:180     theta=theta+dth;     aint1=aint1+(dth*gchi1(theta,ak,qTF2));		%call function aint1  end	y1(i)=const*aint1/ak3;   x1(i)=ei*1.d3; end figure(1);hold on;plot(x1,y1,'r');axis([0,emax*1.d3,0,16e12]);end;xlabel('Energy, E (meV)');ylabel('Elastic scattering rate, 1/tau (s-1)');ttl=sprintf('Chapt8Fig14a, nmax=%4.1e cm-3, m*=%5.2f, epsilon0r=%5.1f',n,m1,ezero);title(ttl);hold off;function gchi1 = gchi1(theta,k,qTF2) x=2.0*k*sin(theta/2.0); x2=x^2;	 g=(tan(theta/2.0))*((sin(theta/2.))^2)*((1.0+(qTF2/x2))^2); gchi1=1.0/g;return

\begin{abstract}
This document provides complete theoretical background and code analysis for \texttt{Chapt8Fig14a.m}, which calculates the total elastic scattering rate due to ionized impurities as a function of electron energy. The code employs Thomas-Fermi screening and numerical angular integration using the trapezoidal rule.
\end{abstract}

%==============================================================================
\section{Theoretical Foundation}
%==============================================================================

\subsection{Ionized Impurity Scattering Theory}

\begin{axiom}[Fermi's Golden Rule for Scattering]
The scattering rate from state $\mathbf{k}$ to state $\mathbf{k}'$ is:
\begin{equation}
W_{\mathbf{k}\to\mathbf{k}'} = \frac{2\pi}{\hbar}|V_{\mathbf{k}\mathbf{k}'}|^2 \delta(E_{\mathbf{k}'} - E_{\mathbf{k}})
\end{equation}
\end{axiom}

\begin{definition}[Screened Coulomb Matrix Element]
For ionized impurity scattering with Thomas-Fermi screening:
\begin{equation}
V_{\mathbf{k}\mathbf{k}'} = \frac{e^2}{\varepsilon_0\varepsilon_r} \cdot \frac{1}{q^2 + q_{\text{TF}}^2}
\end{equation}
where $q = |\mathbf{k}' - \mathbf{k}|$ is the momentum transfer.
\end{definition}

\subsection{Total Scattering Rate}

\begin{theorem}[Total Elastic Scattering Rate]
The total scattering rate is obtained by integrating over all final states:
\begin{equation}
\frac{1}{\tau(E)} = \sum_{\mathbf{k}'} W_{\mathbf{k}\to\mathbf{k}'} = n_i \int |V(q)|^2 g(\mathbf{k}') \delta(E' - E) \, d^3k'
\end{equation}
where $n_i$ is the impurity concentration.
\end{theorem}

\begin{theorem}[Angular Integration Form]
For elastic scattering on a spherical Fermi surface:
\begin{equation}
\frac{1}{\tau(E)} = \frac{\pi n_i m^*}{\hbar^3 k^3} \left(\frac{e^2}{4\pi\varepsilon_0\varepsilon_r}\right)^2 \int_0^\pi \frac{\sin\theta}{(q^2 + q_{\text{TF}}^2)^2} \, d\theta
\end{equation}
where $q = 2k\sin(\theta/2)$.
\end{theorem}

\subsection{Change of Variables}

\begin{lemma}[Substitution to $\eta$]
With $\eta = \sin(\theta/2)$:
\begin{itemize}
\item $q = 2k\eta$
\item $d\eta = \frac{1}{2}\cos(\theta/2)\,d\theta$
\item $\sin\theta = 2\sin(\theta/2)\cos(\theta/2) = 2\eta\sqrt{1-\eta^2}$
\end{itemize}
\end{lemma}

The integral becomes:
\begin{equation}
\int_0^\pi \frac{\sin\theta}{(q^2 + q_{\text{TF}}^2)^2} \, d\theta = \int_0^1 \frac{4\eta\sqrt{1-\eta^2}}{(4k^2\eta^2 + q_{\text{TF}}^2)^2} \cdot \frac{2d\eta}{\sqrt{1-\eta^2}}
\end{equation}

\subsection{Thomas-Fermi Screening}

\begin{definition}[Thomas-Fermi Wavevector]
\begin{equation}
q_{\text{TF}}^2 = \frac{k_F m^* e^2}{\pi^2 \hbar^2 \varepsilon_0 \varepsilon_r}
\end{equation}
where $k_F = (3\pi^2 n)^{1/3}$ is the Fermi wavevector.
\end{definition}

%==============================================================================
\section{Numerical Method: Trapezoidal Integration}
%==============================================================================

\subsection{The Trapezoidal Rule}

\begin{theorem}[Trapezoidal Rule]
For a function $f(x)$ on interval $[a, b]$ with $N$ subintervals of width $h = (b-a)/N$:
\begin{equation}
\int_a^b f(x)\,dx \approx h\left[\frac{f(a) + f(b)}{2} + \sum_{i=1}^{N-1} f(a + ih)\right]
\end{equation}
\end{theorem}

\begin{theorem}[Error Estimate]
The error in the trapezoidal rule is:
\begin{equation}
E_T = -\frac{(b-a)^3}{12N^2} f''(\xi)
\end{equation}
for some $\xi \in [a, b]$. The method is second-order accurate: $E_T = O(h^2)$.
\end{theorem}

\subsection{Application to Angular Integration}

The code uses a simplified form where the step size equals the increment:
\begin{equation}
\int_0^\pi f(\theta)\,d\theta \approx \sum_{j=1}^{180} f(\theta_j) \cdot \Delta\theta
\end{equation}
where $\Delta\theta = \pi/180$ radians (1¬∞ steps).

\section{Physical Interpretation}
%==============================================================================

\subsection{Energy Dependence}

The scattering rate $1/\tau(E)$ typically:
\begin{itemize}
\item Decreases with increasing energy (faster electrons harder to scatter)
\item Scales approximately as $E^{-3/2}$ at high energy
\item Has a plateau or peak near $E_F$
\end{itemize}

\subsection{Density Dependence}

Higher carrier density leads to:
\begin{itemize}
\item Stronger screening (larger $q_{\text{TF}}$)
\item Lower scattering rate at small angles
\item But higher overall scattering due to more impurities (if $n_i = n$)
\end{itemize}

%==============================================================================
\section{Summary}
%==============================================================================

This code calculates the total elastic scattering rate by numerically integrating over all scattering angles using the trapezoidal rule. The helper function \texttt{gchi1} encodes the screened Coulomb scattering matrix element with Thomas-Fermi screening. Results show how scattering rate varies with electron energy and carrier density.
