\section{Theoretical Foundation}
%==============================================================================

\subsection{Bare Coulomb Scattering}

\begin{theorem}[Rutherford Scattering Rate]
For bare Coulomb scattering (no screening):
\begin{equation}
\frac{d(1/\tau)}{d\theta} = \frac{2\pi m^*}{\hbar^3 k^3} n_i \left(\frac{e^2}{4\pi\varepsilon_0\varepsilon_r}\right)^2 \frac{\sin\theta}{q^4}
\end{equation}
where $q = 2k\sin(\theta/2)$.
\end{theorem}

\begin{corollary}[Small-Angle Divergence]
As $\theta \to 0$:
\begin{equation}
\frac{d(1/\tau)}{d\theta} \propto \frac{\theta}{(k\theta)^4} = \frac{1}{k^4\theta^3} \to \infty
\end{equation}
This divergence is unphysical and leads to an infinite total scattering rate.
\end{corollary}

\subsection{Screened Coulomb Scattering}

\begin{theorem}[Thomas-Fermi Screened Rate]
With Thomas-Fermi screening:
\begin{equation}
\frac{d(1/\tau)}{d\theta} = \frac{2\pi m^*}{\hbar^3 k^3} n_i \left(\frac{e^2}{4\pi\varepsilon_0}\right)^2 \frac{\sin\theta}{\varepsilon_{\text{TF}}^2(q) \cdot q^4}
\end{equation}
where:
\begin{equation}
\varepsilon_{\text{TF}}(q) = \varepsilon_r\left(1 + \frac{q_{\text{TF}}^2}{q^2}\right)
\end{equation}
\end{theorem}

\begin{corollary}[Regularization]
As $\theta \to 0$ with screening:
\begin{equation}
\frac{d(1/\tau)}{d\theta} \propto \frac{\theta}{(q_{\text{TF}}^2)^2} = \text{finite}
\end{equation}
The screening wavevector $q_{\text{TF}}$ provides a natural cutoff.
\end{corollary}

\subsection{Physical Origin of Screening}

\begin{definition}[Debye-H\"uckel/Thomas-Fermi Picture]
Mobile electrons redistribute around a charged impurity:
\begin{equation}
\delta n(r) \propto e^{-q_{\text{TF}} r}
\end{equation}
This creates a screening cloud that reduces the effective range of the Coulomb potential.
\end{definition}

The screened potential in real space:
\begin{equation}
V(r) = \frac{Ze^2}{4\pi\varepsilon_0\varepsilon_r r} e^{-q_{\text{TF}} r}
\end{equation}

This Yukawa-type potential has a finite range $\sim 1/q_{\text{TF}}$.

\left(\frac{1}{\tau}\right)_{\text{bare}} = \frac{2\pi m^*}{\hbar^3 k^3} n \left(\frac{e^2}{4\pi}\right)^2 \frac{d\eta}{\varepsilon^2 \eta^3}
\end{equation}
\textit{Bare Coulomb scattering rate (no screening, only background dielectric constant).}

\subsection{Thomas-Fermi Screened Rate}

\begin{lstlisting}

\left(\frac{1}{\tau}\right)_{\text{TF}} = \frac{2\pi m^*}{\hbar^3 k^3} n \left(\frac{e^2}{4\pi}\right)^2 \frac{d\eta}{\varepsilon_{\text{TF}}^2 \eta^3}
\end{equation}
\textit{Thomas-Fermi screened scattering rate.}

\textit{Red: Thomas-Fermi screened; Blue: Bare Coulomb (unscreened).}

\section{Physical Interpretation}
%==============================================================================

\subsection{Comparison of Curves}

\begin{itemize}
\item \textbf{Blue curve (unscreened)}: Diverges at small angles
\item \textbf{Red curve (TF screened)}: Finite at all angles
\item At large angles ($\theta \gtrsim 90¬∞$): Both curves similar
\item At small angles ($\theta \lesssim 30¬∞$): Dramatic difference
\end{itemize}

\subsection{Crossover Angle}

The screening becomes important when $q \sim q_{\text{TF}}$:
\begin{equation}
2k\sin(\theta_c/2) \approx q_{\text{TF}}
\end{equation}
\begin{equation}
\theta_c \approx 2\arcsin\left(\frac{q_{\text{TF}}}{2k}\right)
\end{equation}

For small angles:
\begin{equation}
\theta_c \approx \frac{q_{\text{TF}}}{k}
\end{equation}

\subsection{Total Scattering Rate}

Without screening:
\begin{equation}
\frac{1}{\tau_{\text{bare}}} = \int_0^\pi \frac{d(1/\tau)}{d\theta} d\theta = \infty
\end{equation}

With Thomas-Fermi screening:
\begin{equation}
\frac{1}{\tau_{\text{TF}}} = \int_0^\pi \frac{d(1/\tau)}{d\theta} d\theta = \text{finite}
\end{equation}

\subsection{Practical Implications}

\begin{itemize}
\item Bare Coulomb formula (Rutherford) cannot be used for transport calculations
\item Screening is essential for finite mobility in doped semiconductors
\item Higher doping $\Rightarrow$ stronger screening $\Rightarrow$ can partially compensate for more scatterers
\end{itemize}

%==============================================================================
\section{Summary}
%==============================================================================

This code provides a direct visual comparison between screened and unscreened Coulomb scattering, demonstrating:

\begin{enumerate}
\item The bare Coulomb potential leads to divergent scattering at small angles
\item Thomas-Fermi screening regularizes this divergence
\item The crossover occurs at angle $\theta_c \sim q_{\text{TF}}/k$
\item At large angles, screening effects are minimal
\end{enumerate}

This is a fundamental result in semiconductor transport theory, explaining why heavily doped semiconductors can still have finite (though reduced) mobility.
