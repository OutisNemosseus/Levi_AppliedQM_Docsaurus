% Chapt7Fig6a.m% plots Fermi function for fixed carrier density and a number of temperatures% carrier density n(cm-3), temperature kelvin(K), effective mass m1, relative error rerr% uses function fermi.m and chempot.mclear;clf;	n=1.e18						%carrier density (cm-3)      m0=9.109382;				%bare electron mass( x10^31 kg)   m1=0.07;						%effective electron mass	echarge=1.6021764;		%electron charge (x10^19 C)	hbar=1.05457159;			%Planck's constant (x10^34 J s)   kB=8.61734e-5;				%Boltzmann constant (eV K^-1)      rerr=1.e-3;					%relative error in calculation of chemical potential   Temperature=[0.1,100,200,300,400];				%Temperature values(K)   nTemp=length(Temperature)							%number of different temperatures   	nplotpoints=300;			%number of data points in each plot   Emin=0;						%minimum energy (meV)   Emax=200;					%maximum energy (meV) 	Estep=(Emax-Emin)/nplotpoints;			%Step in energy (meV)    for j=1:1:nTemp   kelvin=Temperature(j);   kBT=1000.*kelvin*kB;   beta=1./kBT;   mu1=chempot(kelvin,m1,n,rerr);  for k = 1:1:nplotpoints+1     energy=((k-1)*Estep)+Emin;     E(k)=energy;     Prob=fermi(beta,energy,mu1);     FD(k)=Prob;  end;    hold on;figure(1);plot(E,log(FD));axis([0 Emax -10 0]);endxlabel('Energy, E (meV)');ylabel('ln(occupation factor)');ttl=sprintf('Chapt7Fig6a, m* = %4.2f m0, Tmax = %4.1f K, n = %7.2e cm-3',m1,kelvin,n);title(ttl);

\section{Theoretical Foundation: From Axioms to Fermi-Dirac Statistics}
%==============================================================================

\subsection{Fundamental Axioms of Quantum Statistical Mechanics}

\begin{axiom}[Indistinguishability of Identical Particles]
In quantum mechanics, identical particles (such as electrons) are fundamentally indistinguishable. The state of a system is described by occupation numbers rather than individual particle trajectories.
\end{axiom}

\begin{axiom}[Pauli Exclusion Principle]
For fermions (particles with half-integer spin, including electrons), no two particles can occupy the same quantum state simultaneously. Mathematically, for occupation number $n_i$ of state $i$:
\begin{equation}
n_i \in \{0, 1\}
\end{equation}
\end{axiom}

\begin{axiom}[Principle of Equal A Priori Probabilities]
In thermal equilibrium, all accessible microstates of an isolated system are equally probable.
\end{axiom}

\begin{axiom}[Boltzmann Entropy Hypothesis]
The entropy $S$ of a system is related to the number of microstates $\Omega$ by:
\begin{equation}
S = k_B \ln \Omega
\end{equation}
where $k_B$ is Boltzmann's constant.
\end{axiom}

\subsection{Derivation of the Fermi-Dirac Distribution}

\begin{definition}[Grand Canonical Ensemble]
For a system in thermal and chemical equilibrium with a reservoir at temperature $T$ and chemical potential $\mu$, the probability of a microstate with energy $E$ and particle number $N$ is:
\begin{equation}
P(E, N) = \frac{1}{\mathcal{Z}} e^{-\beta(E - \mu N)}
\end{equation}
where $\beta = 1/(k_B T)$ and $\mathcal{Z}$ is the grand partition function.
\end{definition}

\begin{theorem}[Fermi-Dirac Distribution Function]
For a system of non-interacting fermions in thermal equilibrium, the average occupation number of a single-particle state with energy $E$ is:
\begin{equation}
f(E) = \frac{1}{e^{\beta(E-\mu)} + 1}
\end{equation}
\end{theorem}

\begin{proof}
Consider a single quantum state with energy $\epsilon$. Due to the Pauli exclusion principle, the occupation number can only be $n = 0$ or $n = 1$.

The grand partition function for this single state is:
\begin{equation}
\mathcal{Z}_1 = \sum_{n=0}^{1} e^{-\beta(\epsilon - \mu)n} = 1 + e^{-\beta(\epsilon - \mu)}
\end{equation}

The average occupation is:
\begin{align}
\langle n \rangle &= \frac{\sum_{n=0}^{1} n \cdot e^{-\beta(\epsilon - \mu)n}}{\mathcal{Z}_1} \\
&= \frac{0 \cdot 1 + 1 \cdot e^{-\beta(\epsilon - \mu)}}{1 + e^{-\beta(\epsilon - \mu)}} \\
&= \frac{e^{-\beta(\epsilon - \mu)}}{1 + e^{-\beta(\epsilon - \mu)}} \\
&= \frac{1}{e^{\beta(\epsilon - \mu)} + 1}
\end{align}
\end{proof}

\subsection{Chemical Potential in Semiconductors}

\begin{definition}[Density of States in 3D]
For a parabolic band with effective mass $m^*$, the density of states per unit volume is:
\begin{equation}
g(E) = \frac{1}{2\pi^2} \left(\frac{2m^*}{\hbar^2}\right)^{3/2} \sqrt{E}
\end{equation}
\end{definition}

\begin{theorem}[Carrier Density Integral]
The electron concentration in the conduction band is given by:
\begin{equation}
n = \int_0^{\infty} g(E) f(E) \, dE = \frac{1}{2\pi^2} \left(\frac{2m^*}{\hbar^2}\right)^{3/2} \int_0^{\infty} \frac{\sqrt{E}}{e^{\beta(E-\mu)} + 1} \, dE
\end{equation}
\end{theorem}

\begin{definition}[Fermi Integral of Order 1/2]
The Fermi integral of order $j$ is defined as:
\begin{equation}
\mathcal{F}_j(\eta) = \frac{1}{\Gamma(j+1)} \int_0^{\infty} \frac{x^j}{e^{x-\eta} + 1} \, dx
\end{equation}
where $\eta = \mu/(k_B T)$ is the reduced chemical potential.
\end{definition}

\begin{theorem}[Carrier Density in Terms of Fermi Integral]
\begin{equation}
n = N_c \mathcal{F}_{1/2}(\eta)
\end{equation}
where $N_c = 2\left(\frac{m^* k_B T}{2\pi\hbar^2}\right)^{3/2}$ is the effective density of states.
\end{theorem}

\subsection{Numerical Method: Newton-Raphson for Chemical Potential}

The code uses a helper function \texttt{chempot.m} to find $\mu$ given $n$. This requires solving the implicit equation:
\begin{equation}
n = N_c \mathcal{F}_{1/2}\left(\frac{\mu}{k_B T}\right)
\end{equation}

\begin{theorem}[Newton-Raphson Method]
To find the root of $f(x) = 0$, the Newton-Raphson iteration is:
\begin{equation}
x_{n+1} = x_n - \frac{f(x_n)}{f'(x_n)}
\end{equation}
\end{theorem}

For finding chemical potential, we solve:
\begin{equation}
F(\mu) = n - N_c \mathcal{F}_{1/2}\left(\frac{\mu}{k_B T}\right) = 0
\end{equation}

The derivative uses the relation:
\begin{equation}
\frac{d\mathcal{F}_j}{d\eta} = \mathcal{F}_{j-1}(\eta)
\end{equation}

\section{Physical Interpretation}
%==============================================================================

\subsection{Temperature Dependence}

\begin{itemize}
\item At $T \to 0$: The Fermi function becomes a step function
\begin{equation}
f(E) = \begin{cases} 1 & E < \mu \\ 0 & E > \mu \end{cases}
\end{equation}

\item At finite $T$: The step is ``smeared'' over an energy range $\sim k_B T$

\item The slope of $\ln f(E)$ in the tail region is $-1/(k_B T)$
\end{itemize}

\subsection{Significance of Parameters}

\begin{itemize}
\item $n = 10^{18}$ cm$^{-3}$: Heavy doping, chemical potential likely positive (degenerate regime)
\item $m^* = 0.07 m_0$: Light effective mass typical of III-V semiconductors (GaAs)
\item Temperature range 0.1--400 K spans quantum to classical regimes
\end{itemize}

%==============================================================================
\section{Summary}
%==============================================================================

This code visualizes the fundamental Fermi-Dirac distribution that governs electron statistics in semiconductors. The logarithmic plot clearly shows the transition from the plateau region ($f \approx 1$ for $E < \mu$) to the exponential tail ($f \propto e^{-E/k_B T}$ for $E \gg \mu$), with the transition width controlled by temperature.
