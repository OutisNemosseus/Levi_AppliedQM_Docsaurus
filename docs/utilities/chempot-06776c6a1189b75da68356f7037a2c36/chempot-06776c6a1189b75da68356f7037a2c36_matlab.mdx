% chempot uses function fermi.m% carrier density n(cm-3), temperature kelvin(K), relative error rerr% returns chemical potential mu1 in meVfunction [chempot]=chempot(kelvin,m1,n,rerr)   m0=9.109382;												%bare electron mass (x10^31 kg)	echarge=1.6021764;										%electron charge (x10^19 C)	hbar=1.05457159;											%Planck's constant (x10^34 J s)   kB=8.61734e-5;												%Boltzmann constant (eV K-1)      kF1=(3.0*(pi^2.0)*n)^(1/3);							%Fermi wave vector (cm-1)   eF=1.e-11*((hbar*kF1)^2)/(2.0*m0*m1*echarge);	%Fermi energy (meV)   kBT=1000.*kelvin*kB;										%thermal energy (meV)   beta=1./kBT;												%inverse thermal energy (meV-1)   	mumax=eF;	x=((n*1.e6)/2.)*(((2.e-15*pi*beta*(hbar^2))/(echarge*m0*m1))^1.5);	mumin=(+1./beta)*log(x);      emax=eF+(15./beta);   de=emax/1000.;	const=1.e16*((20.*echarge*m1*m0)^.5)*echarge*m1*m0/((pi^2)*(hbar^3));   for j=1:25   mu1=mumin+((mumax-mumin)/2.);	energy=0.0;	ainter=0.0;   %	calculate carrier density n'	for i=1:1000;		energy=energy+de;		ainter=ainter+(((sqrt(energy))*de)*fermi(beta,energy,mu1));   end;   nprime=const*ainter;      delta=(n-nprime)/n;										%  delta is relative error in carrier density   	if((abs(delta)) < rerr)      	break;     			elseif(delta < 0.)          	mumax=mu1;			else 			mumin=mu1;   	end;end;if j >= 25   'check convergence!'end;chempot=mu1;return
